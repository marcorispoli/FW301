534__attribute__((always_inline)) __STATIC_INLINE uint32_t __RBIT(uint32_t value)
535{
536 uint32_t result;
537 uint32_t s = (4U /*sizeof(v)*/ * 8U) - 1U; /* extra shift needed at end */
538
539 result = value; /* r will be reversed bits of v; first get LSB of v */
540 for (value >>= 1U; value != 0U; value >>= 1U)
541 {
542 result <<= 1U;
543 result |= value & 1U;
544 s--;
545 }
546 result <<= s; /* shift when v's highest bits are zero */
547 return result;
548}
549#endif

754__attribute__((always_inline)) __STATIC_INLINE int32_t __SSAT(int32_t val, uint32_t sat)
755{
756 if ((sat >= 1U) && (sat <= 32U))
757 {
758 const int32_t max = (int32_t)((1U << (sat - 1U)) - 1U);
759 const int32_t min = -1 - max ;
760 if (val > max)
761 {
762 return max;
763 }
764 else if (val < min)
765 {
766 return min;
767 }
768 }
769 return val;
770}
771
779__attribute__((always_inline)) __STATIC_INLINE uint32_t __USAT(int32_t val, uint32_t sat)
780{
781 if (sat <= 31U)
782 {
783 const uint32_t max = ((1U << sat) - 1U);
784 if (val > (int32_t)max)
785 {
786 return max;
787 }
788 else if (val < 0)
789 {
790 return 0U;
791 }
792 }
793 return (uint32_t)val;
794}
